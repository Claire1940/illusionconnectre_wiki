  <Step num="1" title="ล็อกอินแบบ Guest">

  </Step>
  <Step num="2" title="รับรางวัล">

  </Step>
  <Step num="3" title="การสุ่มแบบเจาะจง">

  </Step>
  <Step num="4" title="ผูกบัญชี">
